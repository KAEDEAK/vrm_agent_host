/* CDK-00210: ローカライズ初期化 & テキスト取得  */
let i18nData = null;
let currentLangData = null;
let selectedLang = 'en';

/**
 * 初期化
 */
async function initializeLocalization() {
  try {
    const resp = await fetch('/localize.json'); // 実際のパスに置き換えてね
    i18nData = await resp.json();
    const userLang = navigator.language || navigator.userLanguage;
    const storedLang = localStorage.getItem('selected_language');
    selectedLang = storedLang || i18nData.data.default || 'en';

    if (!storedLang && userLang.indexOf('ja') !== -1) {
      selectedLang = 'ja';
    }

    updateCurrentLangData(selectedLang);
  } catch (e) {
    console.error('Localization init failed:', e);
  }
}

/**
 * メッセージを取得
 * @param {string} msgId - localize.json 内の ID
 * @returns {string}
 */
function getLocalizedText(msgId) {
  if (!currentLangData) {
    return 'Localization not loaded.';
  }
  const arr = currentLangData.msgs;
  const found = arr.find(o => o.id === msgId);
  if (!found) {
    return `[${msgId}] missing`;
  }
  return found.txt;
}

/**
 * HTML要素へ反映
 * data-locale 属性に ID を書いてある要素に対し、テキストを置き換える
 */
function applyLocalizationToPage() {
  document.querySelectorAll('[data-locale]').forEach(elem => {
    const id = elem.dataset.locale;
    if (!id) return;
    const txt = getLocalizedText(id);
    elem.textContent = txt;
  });

  // placeholder 用
  document.querySelectorAll('[data-locale-placeholder]').forEach(elem => {
    const id = elem.dataset.localePlaceholder;
    if (!id) return;
    const txt = getLocalizedText(id);
    if (elem.tagName === 'TEXTAREA' || elem.tagName === 'INPUT') {
      elem.placeholder = txt;
    }
  });

  // ボタンの value 用
  document.querySelectorAll('[data-locale-value]').forEach(elem => {
    const id = elem.dataset.localeValue;
    if (!id) return;
    const txt = getLocalizedText(id);
    if (elem.tagName === 'INPUT') {
      elem.value = txt;
    } else if (elem.tagName === 'BUTTON') {
      elem.textContent = txt;
    }
  });


  
}

function updateCurrentLangData(lang) {
  const langs = i18nData.data.langs;
  currentLangData = langs.find(l => l.lang === lang);
  if (!currentLangData) {
    currentLangData = langs.find(l => l.lang === 'en');
  }
  selectedLang = lang;
  localStorage.setItem('selected_language', lang);
}

function changeLanguage(lang) {
  updateCurrentLangData(lang);
  applyLocalizationToPage();
}

/**
 * 全体の呼び出し例
 */
async function initI18nAndApply() {
  await initializeLocalization();
  applyLocalizationToPage();
  const langSelect = document.getElementById('languageSelect');
  if (langSelect) {
    langSelect.value = selectedLang;
  }
  console.log("Localization applied!");
      const tocList = document.getElementById('tocList');
      document.querySelectorAll('h2').forEach((h2, idx) => {
        const title = h2.textContent.trim();
        const anchorId = 'section_' + idx;
        h2.id = anchorId;

      const li = document.createElement('li');
      const a = document.createElement('a');
      a.href = '#' + anchorId;
      a.textContent = title;
      li.appendChild(a);
      tocList.appendChild(li);
    });
}

window.changeLanguage = changeLanguage;
